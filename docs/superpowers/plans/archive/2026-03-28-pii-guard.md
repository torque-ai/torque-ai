# PII Guard Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Three-layer auto-sanitization that silently replaces personal data (user paths, private IPs, emails, hostnames, custom patterns) before it reaches the repository.

**Architecture:** A shared pattern engine (`server/utils/pii-guard.js`) loaded by three layers: TORQUE output safeguards (post-task), Claude Code PreToolUse hook (before Write/Edit), and a git pre-commit hook (before commit). Configuration stored in TORQUE's `project_config` table via `set_project_defaults`.

**Tech Stack:** Node.js, regex patterns, TORQUE REST API, bash/shell for git hook, Claude Code hooks JSON

```js
// server/utils/pii-guard.js
'use strict';

const os = require('os');

// Auto-detect system hostname at module load
const SYSTEM_HOSTNAME = os.hostname();

// Built-in PII categories
const BUILTIN_CATEGORIES = {
  user_paths: {
    patterns: [
      { regex: /C:\\Users\\([^\\]+)/g, replacement: 'C:\\Users\\<user>' },
      { regex: /\/home\/([^/\s]+)/g, replacement: '/home/<user>' },
      { regex: /\/Users\/([^/\s]+)/g, replacement: '/Users/<user>' },
    ]
  },
  private_ips: {
    patterns: [
      { regex: /192\.168\.\d+\.(\d+)/g, replacement: '192.0.2.$1' },
      { regex: /\b10\.\d+\.\d+\.(\d+)\b/g, replacement: '10.0.0.$1' },
      { regex: /\b172\.(1[6-9]|2\d|3[01])\.\d+\.(\d+)\b/g, replacement: '172.16.0.$2' },
    ]
  },
  emails: {
    patterns: [
      {
        regex: /[a-zA-Z0-9._%+-]+@(?!example\.com\b)(?!test\.com\b)(?!noreply\b)[a-zA-Z0-9.-]+\.[a-zA-Z]{2,}/g,
        replacement: 'user@example.com'
      }
    ]
  },
  hostnames: {
    // Populated dynamically from os.hostname()
    patterns: []
  }
};

// Populate hostname pattern if hostname is non-trivial (>2 chars)
if (SYSTEM_HOSTNAME && SYSTEM_HOSTNAME.length > 2) {
  const escaped = SYSTEM_HOSTNAME.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
  BUILTIN_CATEGORIES.hostnames.patterns.push({
    regex: new RegExp(escaped, 'gi'),
    replacement: 'example-host'
  });
}

/**
 * Scan text for PII and return sanitized version with findings.
 *
 * @param {string} text - Text to scan
 * @param {object} [options]
 * @param {object} [options.builtinOverrides] - { category: false } to disable specific categories
 * @param {Array} [options.customPatterns] - [{ pattern, replacement, regex? }]
 * @returns {{ clean: boolean, sanitized: string, findings: Array<{ category: string, match: string, line: number }> }}
 */
function scanAndReplace(text, options = {}) {
  if (!text || typeof text !== 'string') {
    return { clean: true, sanitized: '', findings: [] };
  }

  const { builtinOverrides = {}, customPatterns = [] } = options;
  const findings = [];
  let sanitized = text;

  // Apply built-in categories
  for (const [category, config] of Object.entries(BUILTIN_CATEGORIES)) {
    if (builtinOverrides[category] === false) continue;

    for (const { regex, replacement } of config.patterns) {
      // Reset lastIndex for global regexes
      regex.lastIndex = 0;

      // Collect findings before replacing
      let match;
      const regexCopy = new RegExp(regex.source, regex.flags);
      while ((match = regexCopy.exec(sanitized)) !== null) {
        const line = sanitized.substring(0, match.index).split('\n').length;
        findings.push({ category, match: match[0], line });
      }

      // Apply replacement
      regex.lastIndex = 0;
      sanitized = sanitized.replace(regex, replacement);
    }
  }

  // Apply custom patterns
  for (const custom of customPatterns) {
    let customRegex;
    if (custom.regex) {
      customRegex = new RegExp(custom.pattern, 'g');
    } else {
      const escaped = custom.pattern.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
      customRegex = new RegExp(escaped, 'g');
    }

    let match;
    const findRegex = new RegExp(customRegex.source, customRegex.flags);
    while ((match = findRegex.exec(sanitized)) !== null) {
      const line = sanitized.substring(0, match.index).split('\n').length;
      findings.push({ category: 'custom', match: match[0], line });
    }

    sanitized = sanitized.replace(customRegex, custom.replacement);
  }

  return {
    clean: findings.length === 0,
    sanitized,
    findings
  };
}

module.exports = { scanAndReplace, BUILTIN_CATEGORIES, SYSTEM_HOSTNAME };
```

```js
  if (req.method === 'POST' && url === '/api/pii-scan') {
    const piiGuard = require('./utils/pii-guard');
    const body = typeof req.body === 'object' ? req.body : {};
    const text = body.text || '';

    let customPatterns = [];

    if (workingDir) {

          if (piiJson) {
```

- [ ] **Step 1: Write the pre-commit hook script**

```bash
#!/usr/bin/env bash
# PII Guard — Git Pre-Commit Hook
# Scans staged files for personal data and auto-fixes before commit.
# Calls TORQUE REST API when available, falls back to built-in regex scan.

set -euo pipefail

TORQUE_API="${TORQUE_API_URL:-http://127.0.0.1:3457}"
PII_ENDPOINT="${TORQUE_API}/api/pii-scan"
WORKING_DIR="$(git rev-parse --show-toplevel)"

# Binary file extensions to skip
BINARY_EXTS="png|jpg|jpeg|gif|bmp|ico|svg|woff|woff2|ttf|eot|mp3|mp4|wav|zip|tar|gz|pdf|db|sqlite|exe|dll|so|dylib"

# Built-in fallback patterns (used when TORQUE is unreachable)
fallback_scan() {
  local file="$1"
  local content
  content=$(cat "$file" 2>/dev/null) || return 0

  local dirty=0

  # User paths
  if echo "$content" | grep -qP 'C:\\Users\\[^\\]+|/home/<user>/\s]+|/Users/<user>/\s]+'; then
    dirty=1
  fi

  # Private IPs
  if echo "$content" | grep -qP '192\.168\.\d+\.\d+|\b10\.\d+\.\d+\.\d+\b|\b172\.(1[6-9]|2[0-9]|3[01])\.\d+\.\d+\b'; then
    dirty=1
  fi

  # Emails (skip example.com, test.com, noreply@)
  if echo "$content" | grep -qP '[a-zA-Z0-9._%+-]+@(?!example\.com|test\.com|noreply)[a-zA-Z0-9.-]+\.[a-zA-Z]{2,}'; then
    dirty=1
  fi

  if [ "$dirty" -eq 1 ]; then
    echo "PII-GUARD [fallback]: PII detected in $file (TORQUE unavailable — cannot auto-fix)"
    echo "  Start TORQUE for auto-fix, or clean manually."
    return 1
  fi
  return 0
}

# Check if TORQUE is reachable
torque_available() {
  curl -s --max-time 2 "${TORQUE_API}/api/version" > /dev/null 2>&1
}

# Scan a file via TORQUE API
torque_scan() {
  local file="$1"
  local content
  content=$(cat "$file" 2>/dev/null) || return 0

  # Skip empty files
  [ -z "$content" ] && return 0

  local response
  response=$(curl -s --max-time 10 \
    -X POST "$PII_ENDPOINT" \
    -H "Content-Type: application/json" \
    -d "$(jq -n --arg text "$content" --arg wd "$WORKING_DIR" \
      '{text: $text, working_directory: $wd}')" \
    2>/dev/null) || return 1

  local clean
  clean=$(echo "$response" | jq -r '.clean' 2>/dev/null) || return 1

  if [ "$clean" = "false" ]; then
    local sanitized
    sanitized=$(echo "$response" | jq -r '.sanitized' 2>/dev/null) || return 1
    local finding_count
    finding_count=$(echo "$response" | jq '.findings | length' 2>/dev/null) || finding_count="?"

    # Write sanitized content back and restage
    printf '%s' "$sanitized" > "$file"
    git add "$file"
    echo "PII-GUARD: Auto-fixed $finding_count finding(s) in $file"
  fi

  return 0
}

# Main
staged_files=$(git diff --cached --name-only --diff-filter=ACM)
[ -z "$staged_files" ] && exit 0

has_errors=0

if torque_available; then
  while IFS= read -r file; do
    # Skip binary files
    ext="${file##*.}"
    if echo "$ext" | grep -qiP "^($BINARY_EXTS)$"; then
      continue
    fi

    # Skip files that don't exist (deleted)
    [ -f "$file" ] || continue

    torque_scan "$file" || {
      echo "PII-GUARD: Failed to scan $file"
      has_errors=1
    }
  done <<< "$staged_files"
else
  echo "PII-GUARD: TORQUE unavailable — running fallback regex scan"
  while IFS= read -r file; do
    ext="${file##*.}"
    if echo "$ext" | grep -qiP "^($BINARY_EXTS)$"; then
      continue
    fi
    [ -f "$file" ] || continue

    fallback_scan "$file" || has_errors=1
  done <<< "$staged_files"
fi

exit $has_errors
```

```bash
#!/usr/bin/env bash
# PII Guard — Claude Code PreToolUse Hook
# Scans Write/Edit tool content for PII and auto-fixes before execution.
#
# Input: JSON on stdin with tool_input containing the content to scan.
# Output: JSON on stdout — if PII found, returns updated tool_input with sanitized content.
#
# Exit codes:
#   0 = allow (possibly with modified input)
#   2 = block (only if explicitly needed)

set -euo pipefail

TORQUE_API="${TORQUE_API_URL:-http://127.0.0.1:3457}"
PII_ENDPOINT="${TORQUE_API}/api/pii-scan"

# Read hook input from stdin
input=$(cat)
tool_name=$(echo "$input" | jq -r '.tool_name // ""')

# Only process Write and Edit tools
if [ "$tool_name" != "Write" ] && [ "$tool_name" != "Edit" ]; then
  exit 0
fi

# Extract the content to scan
if [ "$tool_name" = "Write" ]; then
  content=$(echo "$input" | jq -r '.tool_input.content // ""')
  working_dir=$(echo "$input" | jq -r '.tool_input.file_path // ""' | sed 's|\\|/|g' | xargs dirname 2>/dev/null || echo "")
elif [ "$tool_name" = "Edit" ]; then
  content=$(echo "$input" | jq -r '.tool_input.new_string // ""')
  working_dir=$(echo "$input" | jq -r '.tool_input.file_path // ""' | sed 's|\\|/|g' | xargs dirname 2>/dev/null || echo "")
fi

# Skip empty content
if [ -z "$content" ]; then
  exit 0
fi

# Try TORQUE API
response=$(curl -s --max-time 5 \
  -X POST "$PII_ENDPOINT" \
  -H "Content-Type: application/json" \
  -d "$(jq -n --arg text "$content" --arg wd "$working_dir" \
    '{text: $text, working_directory: $wd}')" \
  2>/dev/null) || {
  # TORQUE unreachable — allow through (git hook is the backstop)
  exit 0
}

clean=$(echo "$response" | jq -r '.clean' 2>/dev/null) || exit 0

if [ "$clean" = "false" ]; then
  sanitized=$(echo "$response" | jq -r '.sanitized' 2>/dev/null) || exit 0
  finding_count=$(echo "$response" | jq '.findings | length' 2>/dev/null) || finding_count="?"

  # Return modified tool input
  if [ "$tool_name" = "Write" ]; then
    echo "$input" | jq --arg s "$sanitized" '.tool_input.content = $s'
  elif [ "$tool_name" = "Edit" ]; then
    echo "$input" | jq --arg s "$sanitized" '.tool_input.new_string = $s'
  fi

  echo "PII-GUARD: Auto-fixed $finding_count finding(s) in $tool_name call" >&2
fi

exit 0
```

# PII Guard

Automatic personal data sanitization for TORQUE-managed projects.

PII Guard scans for personal data and auto-replaces it with safe placeholders:

set_project_defaults({

## Three Layers

1. **TORQUE Output Safeguards** — scans task output before files are written (automatic, no setup needed)
